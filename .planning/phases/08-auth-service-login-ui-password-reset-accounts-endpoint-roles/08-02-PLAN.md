```typescript
import { Injectable, signal, inject } from '@angular/core';
import { UserOrganizationsService } from '../../api';  // check exact export name

@Injectable({ providedIn: 'root' })
export class AccountStateService {
  private userOrganizationsService = inject(UserOrganizationsService);

  hasConflicts = signal(false);

  refresh(): void {
    this.userOrganizationsService.getMyOrganizations().subscribe({
      next: (response) => {
        const conflicts = response.organizations?.some(org => org.hasConflicts) ?? false;
        this.hasConflicts.set(conflicts);
      },
      error: (error) => {
        console.error('Failed to refresh account state:', error);
      }
    });
  }
}
```

   ```typescript
   private mapErrorToUserMessage(error: HttpErrorResponse): string | null {
     if (error.status === 0) {
       // Network error
       return 'Unable to connect to server. Check your connection and try again.';
     }
     switch (error.status) {
       case 403:
         return "You don't have permission to access this organization.";
       case 410:
         return 'This link has expired. Please request a new one.';
       case 409:
         return 'This action cannot be completed. Please refresh and try again.';
       case 401:
         // Redirect to login will be handled by authGuard
         return null;
       default:
         if (error.status >= 500) {
           return 'Something went wrong. Please try again or contact support.';
         }
         return null; // No user-facing message for other 4xx errors
     }
   }
   ```

   ```typescript
   onInvite(): void {
     this.memberManagementService.inviteMember(
       this.orgContextService.getSettingsOrg(),

     ).subscribe({
       next: (response) => {
         this.errorNotificationService.showSuccess('Member invited successfully');

       }
     });

       case 409:
         return 'User is already a member of this organization.';

     }
   }
   ```

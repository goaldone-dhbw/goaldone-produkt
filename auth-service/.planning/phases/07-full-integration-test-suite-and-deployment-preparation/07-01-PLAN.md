   package de.goaldone.authservice.support;

   import com.fasterxml.jackson.databind.ObjectMapper;
   import org.springframework.beans.factory.annotation.Autowired;
   import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
   import org.springframework.boot.test.context.SpringBootTest;
   import org.springframework.test.context.ActiveProfiles;
   import org.springframework.test.context.DynamicPropertyRegistry;
   import org.springframework.test.context.DynamicPropertySource;
   import org.springframework.test.web.servlet.MockMvc;
   import org.testcontainers.containers.PostgreSQLContainer;
   import org.testcontainers.junit.jupiter.Container;
   import org.testcontainers.junit.jupiter.Testcontainers;

   /**
    * Base class for all integration tests using @SpringBootTest with TestContainers.
    * 
    * Provides:
    * - PostgreSQL container managed by TestContainers
    * - Automatic Liquibase migration execution before tests
    * - Autowired MockMvc and ObjectMapper for API testing
    * - Transaction-based test isolation (each test rolls back)
    * 
    * Usage:
    * <pre>
    * public class YourIntegrationTest extends IntegrationTestBase {
    *     @Test
    *     void testSomeFlow() throws Exception {
    *         mockMvc.perform(get("/api/endpoint"))
    *             .andExpect(status().isOk());
    *     }
    * }
    * </pre>
    */
   @SpringBootTest(
       webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
       properties = {
           "spring.test.mockmvc.print=false",
           "spring.jpa.hibernate.ddl-auto=validate"
       }
   )
   @AutoConfigureMockMvc
   @Testcontainers
   @ActiveProfiles("integration-test")
   public abstract class IntegrationTestBase {

       /**
        * PostgreSQL 16 Alpine container shared across test methods in the same class.
        * 
        * withReuse(true) enables container reuse across test methods for speed.
        * Test isolation is guaranteed by @Transactional on test methods.
        */
       @Container
       static PostgreSQLContainer<?> postgres = new PostgreSQLContainer<>("postgres:16-alpine")
           .withDatabaseName("auth_test")
           .withUsername("test")
           .withPassword("test")
           .withReuse(false);  // Set to false for strict isolation; true for speed

       /**
        * Injects PostgreSQL connection details into Spring context at runtime.
        * This method is called by TestContainers before @SpringBootTest context initialization.
        */
       @DynamicPropertySource
       static void configureTestDatabase(DynamicPropertyRegistry registry) {
           registry.add("spring.datasource.url", postgres::getJdbcUrl);
           registry.add("spring.datasource.username", postgres::getUsername);
           registry.add("spring.datasource.password", postgres::getPassword);
           registry.add("spring.datasource.driver-class-name", () -> "org.postgresql.Driver");
       }

       /**
        * MockMvc instance for performing HTTP requests against the application.
        * 
        * Usage:
        * <pre>
        * mockMvc.perform(post("/api/users")
        *     .contentType(MediaType.APPLICATION_JSON)
        *     .content("{...}"))
        *     .andExpect(status().isCreated());
        * </pre>
        */
       @Autowired
       protected MockMvc mockMvc;

       /**
        * ObjectMapper instance for JSON serialization/deserialization in tests.
        * 
        * Usage:
        * <pre>
        * String json = objectMapper.writeValueAsString(object);
        * MyObject obj = objectMapper.readValue(json, MyObject.class);
        * </pre>
        */
       @Autowired
       protected ObjectMapper objectMapper;
   }
   ```
4. Verify file is created and valid Java (no syntax errors)
5. Check indentation is consistent (use 4 spaces per indent level)

   package de.goaldone.authservice.support;

   import org.junit.jupiter.api.Test;
   import org.springframework.beans.factory.annotation.Autowired;
   import org.springframework.jdbc.core.JdbcTemplate;
   import org.springframework.test.context.jdbc.Sql;

   import static org.assertj.core.api.Assertions.assertThat;

   /**
    * Smoke test to verify integration test infrastructure is working correctly.
    * 
    * This test verifies:
    * - TestContainers PostgreSQL container starts successfully
    * - Liquibase migrations execute and schema is initialized
    * - Application context loads with integration-test profile
    * - Database is accessible via JdbcTemplate
    */
   public class IntegrationTestInfrastructureTest extends IntegrationTestBase {

       @Autowired
       private JdbcTemplate jdbcTemplate;

       @Test
       void testPostgreSQLContainerIsRunning() {
           // If this test passes, PostgreSQL container is running and accessible
           assertThat(jdbcTemplate).isNotNull();
       }

       @Test
       void testLiquibaseSchemaIsInitialized() {
           // Verify that Liquibase migrations have run by checking for a known table
           // The SPRING_USER table should exist (created by Liquibase)
           String query = "SELECT EXISTS(SELECT 1 FROM information_schema.tables " +
               "WHERE table_name = 'spring_user')";
           
           Boolean tableExists = jdbcTemplate.queryForObject(query, Boolean.class);
           assertThat(tableExists).isTrue();
       }

       @Test
       void testDatabaseIsAccessibleAndEmpty() {
           // Verify we can query the database and get a result
           String query = "SELECT COUNT(*) FROM spring_user";
           Integer userCount = jdbcTemplate.queryForObject(query, Integer.class);
           
           assertThat(userCount).isNotNull();
           // Initially empty
           assertThat(userCount).isEqualTo(0);
       }
   }
   ```

## Notes

This plan establishes the foundation for all integration tests in Phase 7 Plan 2. The IntegrationTestBase class is intentionally abstract and lightweight to keep setup simple. Subclasses will add specific test logic (user/org/invitation/password/account linking flows).

**Key Design Decisions Implemented:**
- `PostgreSQLContainer::withReuse(false)` for strict test isolation (can be changed to true for speed if needed)
- `RANDOM_PORT` environment prevents port conflicts in CI/CD
- `DynamicPropertyRegistry` injection handles container connection details automatically
- `@ActiveProfiles("integration-test")` ensures only integration-test.yaml is loaded
- PostgreSQL 16-alpine balances image size and production parity

**Next Steps (Plan 2):** Use IntegrationTestBase to write comprehensive integration tests for all critical flows.

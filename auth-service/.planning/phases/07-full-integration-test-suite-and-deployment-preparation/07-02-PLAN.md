        * Create a company with the given name.
        * Defaults: slug=auto-generated from name, superAdminRequired=true
        */
       public CompanyBuilder createCompany(String name) {
           return new CompanyBuilder(name);
       }

        * Create a company with specific slug.
        */
       public CompanyBuilder createCompanyWithSlug(String name, String slug) {
           return new CompanyBuilder(name).withSlug(slug);
       }

       // ==================== Membership Creation ====================

       /**
        * Create a membership linking user to company with specified role.
        */
       public MembershipBuilder createMembership(User user, Company company, MembershipRole role) {
           return new MembershipBuilder(user, company, role);
       }

       public class UserBuilder {
           private String primaryEmail;
           private String password = "testpass123";
           private boolean emailVerified = false;

           UserBuilder(String primaryEmail) {
               this.primaryEmail = primaryEmail;
           }

           public UserBuilder withPassword(String password) {
               this.password = password;
               return this;
           }

           public UserBuilder withEmailVerified(boolean verified) {
               this.emailVerified = verified;
               return this;
           }

           public User build() {
               User user = new User();
               user.setId(java.util.UUID.randomUUID());
               
               Email primaryEmailEntity = new Email();
               primaryEmailEntity.setEmail(primaryEmail);
               primaryEmailEntity.setVerified(emailVerified);
               primaryEmailEntity.setUser(user);
               
               user.setPrimaryEmail(primaryEmailEntity);
               user.getEmails().add(primaryEmailEntity);
               user.setPassword(passwordEncoder.encode(password));
               
               return userRepository.save(user);
           }
       }

       // ==================== Company Builder ====================

       public class CompanyBuilder {
           private String name;
           private String slug;

           CompanyBuilder(String name) {
               this.name = name;
               this.slug = slugify(name);
           }

           public CompanyBuilder withSlug(String slug) {
               this.slug = slug;
               return this;
           }

           public Company build() {

               company.setId(java.util.UUID.randomUUID());
               company.setName(name);
               company.setSlug(slug);
               
               return companyRepository.save(company);
           }

           private String slugify(String name) {
               return name.toLowerCase()
                   .replaceAll("\\s+", "-")
                   .replaceAll("[^a-z0-9-]", "");
           }
       }

       // ==================== Membership Builder ====================

       public class MembershipBuilder {

           MembershipBuilder(User user, Company company, MembershipRole role) {

               membership.setUser(user);
               membership.setCompany(company);
               membership.setRole(role);
               
               return membershipRepository.save(membership);
           }
       }

           // Given: test OAuth client (registered in Liquibase)

               .andReturn();

           String newAccessToken = objectMapper.readTree(refreshedResponse)
               .get("access_token").asText();
